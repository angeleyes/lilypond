% stacked horizontal lines 
\def\lines#1#2#3{%
 \vbox{\kern-\interstaffrule
        \n=0\nointerlineskip%
        \loop\ifnum\n<#1\advance\n by1%
                \kern\interstaffrule
                \nointerlineskip
                \vbox to 0pt{\hrule height #3 width#2%
                        \vss}\nointerlineskip
                \repeat
        }}
