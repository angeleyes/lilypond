\def\musixcalc{
	\interstaffrule=\balkhoog
	\advance\interstaffrule by-\staffrulethickness
	\divide\interstaffrule by 4
	\advance\interstaffrule by -\staffrulethickness
}

\def\lines#1#2{%
 \vbox{\kern-\interstaffrule
	\n=0\nointerlineskip%
	\loop\ifnum\n<#1\advance\n by1%
		\kern\interstaffrule
		\hrule height \staffrulethickness width#2
		\repeat
	}}
