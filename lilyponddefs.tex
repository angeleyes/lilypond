%
% a staffsymbol with #1 lines, width #2
% bottom at baseline
\def\linestafsym#1#2{\hbox to 0pt%
  {\vbox to 0pt{\vss%
	\vskip-\interlinedist
	\n=0%
	\loop\ifnum\n<#1\advance\n by1%
		\vskip\interstaffrule \hrule height \staffrulethickness	width#2
		\repeat
	}%
\hss}}
