\def\listfont#1#2#3{
        \n=#2
        \advance\n by-1
        \advance\n by-\charsperline
        \loop\ifnum\n<#3
        \advance\n by\charsperline
        \i=0
        {\vbox to\charheight{\vss\centerline{
        {\loop\ifnum\i<\charsperline
        \advance\i by1
        \advance\n by1
        {\setchar{#1}{\number\n}}
        \repeat}}}}\repeat}

\def\title#1{\centerline{\titlefont #1}}
\def\author#1{\centerline{#1}}
